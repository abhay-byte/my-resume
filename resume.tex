% Custom commands
\newcommand{\resumeItem}[1]{
  \item\small{ % List items (like descriptions) are small
    {#1 \vspace{-2pt}}
  }
}

\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
  \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{#1} & \textit{\small #2} \\ % Main subheading bold (default size), secondary small italic
    \textit{\small#3} & \textit{\small #4} \\ % Tertiary info small italic
  \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\ % Sub-subheading items small italic
    \end{tabular*}\vspace{-7pt}
}

    \end{tabular*}\vspace{-7pt}
}
